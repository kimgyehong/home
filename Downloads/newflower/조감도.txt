# 꽃 다이어리 앱 완전 구현 가이드

## 📱 요구사항 분석

### 핵심 기능
1. **동영상 오프닝** → 스킵 가능한 인트로 영상
2. **타이틀 화면** → 아무 곳이나 클릭하면 메인으로 이동
3. **일기장** → 탄생화 배경 + 꾸미기 기능 (글자 색상, 폰트 변경)
4. **도감** → 매일 해금되는 탄생화 컬렉션 + 상세 팝업
5. **기록 보관소** → SQLDelight 기반 일기 저장
6. **BGM 기능** → 4가지 배경음악 선택
7. **랜덤 로딩 화면** → 7개 이미지 중 랜덤 선택 + 톱니바퀴 애니메이션

```kotlin
// ui/android/CollectionScreen.kt
@Composable
fun CollectionScreen(
    viewModel: CollectionViewModel = koinViewModel()
) {
    val flowers by viewModel.flowers.collectAsState()
    val todayFlower by viewModel.todayFlower.collectAsState()
    
    LazyVerticalGrid(
        columns = GridCells.Fixed(3),
        contentPadding = PaddingValues(FlowerDiaryDimens.Padding.Medium)
    ) {
        items(flowers) { flower ->
            FlowerCard(
                flower = flower,
                onClick = { 
                    if (flower.unlocked) {
                        viewModel.showFlowerDetail(flower)
                    }
                }
            )
        }
    }
    
    // 오늘의 꽃 알림
    todayFlower?.let { flower ->
        TodayFlowerNotification(flower = flower)
    }
}

@Composable
fun FlowerCard(
    flower: BirthFlower,
    onClick: () -> Unit
) {
    Card(
        modifier = Modifier
            .padding(FlowerDiaryDimens.Padding.Small)
            .noRippleClickable { onClick() }, // 투명 배경 클릭 문제 해결
        colors = CardDefaults.cardColors(
            containerColor = if (flower.unlocked) Color.White else Color.Gray
        )
    ) {
        Column(
            modifier = Modifier.padding(FlowerDiaryDimens.Padding.Small),
            horizontalAlignment = Alignment.CenterHorizontally
        ) {
            if (flower.unlocked) {
                AsyncImage(
                    model = flower.imageUrl,
                    contentDescription = flower.nameKr,
                    modifier = Modifier.size(80.dp)
                )
            } else {
                Icon(
                    Icons.Default.Lock,
                    contentDescription = "잠금됨",
                    modifier = Modifier.size(80.dp),
                    tint = Color.Gray
                )
            }
            
            Text(
                text = flower.nameKr,
                style = MaterialTheme.typography.bodySmall,
                color = if (flower.unlocked) Color.Black else Color.Gray
            )
        }
    }
}
```

```kotlin
// 투명 배경 클릭 문제 해결을 위한 확장 함수
fun Modifier.noRippleClickable(onClick: () -> Unit): Modifier = composed {
    clickable(
        indication = null,
        interactionSource = remember { MutableInteractionSource() }
    ) {
        onClick()
    }
}
```

### 5. **BGM 시스템**
```kotlin
// platform/android/AndroidAudioManager.kt
actual class AndroidAudioManager(
    private val context: Context
) : AudioManager {
    
    private var mediaPlayer: MediaPlayer? = null
    private val bgmTracks = listOf(
        R.raw.bgm_spring,
        R.raw.bgm_summer,
        R.raw.bgm_autumn,
        R.raw.bgm_winter
    )
    
    actual override fun playBGM(trackIndex: Int) {
        stopBGM()
        
        mediaPlayer = MediaPlayer.create(context, bgmTracks[trackIndex]).apply {
            isLooping = true
            start()
        }
    }
    
    actual override fun stopBGM() {
        mediaPlayer?.apply {
            stop()
            release()
        }
        mediaPlayer = null
    }
    
    actual override fun setBGMVolume(volume: Float) {
        mediaPlayer?.setVolume(volume, volume)
    }
}

// core/common/AudioManager.kt
expect class AudioManager {
    fun playBGM(trackIndex: Int)
    fun stopBGM()
    fun setBGMVolume(volume: Float)
}
```

```kotlin
// feature/diary/DiaryEditorViewModel.kt
class DiaryEditorViewModel(
    private val diaryRepository: DiaryRepository
) : ViewModel() {
    
    private val _selectedFontFamily = MutableStateFlow("default")
    val selectedFontFamily: StateFlow = _selectedFontFamily.asStateFlow()
    
    private val _selectedFontColor = MutableStateFlow(Color.Black)
    val selectedFontColor: StateFlow = _selectedFontColor.asStateFlow()
    
    private val _backgroundFlower = MutableStateFlow(null)
    val backgroundFlower: StateFlow = _backgroundFlower.asStateFlow()
    
    fun changeFontFamily(fontFamily: String) {
        _selectedFontFamily.value = fontFamily
    }
    
    fun changeFontColor(color: Color) {
        _selectedFontColor.value = color
    }
    
    fun setBackgroundFlower(flower: BirthFlower) {
        _backgroundFlower.value = flower
    }
}

// ui/android/DiaryEditorScreen.kt
@Composable
fun DiaryEditorScreen(
    viewModel: DiaryEditorViewModel = koinViewModel()
) {
    val fontFamily by viewModel.selectedFontFamily.collectAsState()
    val fontColor by viewModel.selectedFontColor.collectAsState()
    val backgroundFlower by viewModel.backgroundFlower.collectAsState()
    
    Box(modifier = Modifier.fillMaxSize()) {
        // 배경 꽃 이미지
        backgroundFlower?.let { flower ->
            AsyncImage(
                model = flower.imageUrl,
                contentDescription = null,
                modifier = Modifier.fillMaxSize(),
                contentScale = ContentScale.Crop,
                alpha = 0.3f
            )
        }
        
        // 일기 작성 영역
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(FlowerDiaryDimens.Padding.Medium)
        ) {
            // 꾸미기 도구바
            DiaryToolbar(
                onFontFamilyChange = viewModel::changeFontFamily,
                onFontColorChange = viewModel::changeFontColor,
                onBackgroundChange = viewModel::setBackgroundFlower
            )
            
            // 일기 입력 필드
            OutlinedTextField(
                value = diaryContent,
                onValueChange = { diaryContent = it },
                modifier = Modifier.fillMaxSize(),
                textStyle = TextStyle(
                    fontFamily = getFontFamily(fontFamily),
                    color = fontColor,
                    fontSize = 16.sp
                ),
                colors = TextFieldDefaults.outlinedTextFieldColors(
                    backgroundColor = Color.White.copy(alpha = 0.8f)
                )
            )
        }
    }
}
```

## 🎯 핵심 해결 포인트

### 1. **투명 배경 클릭 문제 해결**
- `indication = null` 설정으로 리플 효과 제거[1]
- `MutableInteractionSource` 사용으로 클릭 상태 제어

### 2. **랜덤 로딩 화면**
- 앱 시작 시 7개 이미지 중 랜덤 선택[2]
- 톱니바퀴 회전 애니메이션 구현

### 3. **365일 탄생화 시스템**
- 매일 자동 해금 시스템 구현[3]
- SQLDelight로 탄생화 데이터 관리

### 4. **BGM 시스템**
- `expect/actual` 패턴으로 플랫폼별 오디오 처리
- 백그라운드 음악 지속 재생

### 5. **일기 꾸미기**
- 폰트 패밀리, 색상 변경 기능
- 탄생화 배경 이미지 적용

이 구조는 **완전한 KMP 호환성**을 보장하며, **Android와 iOS 모두에서 동일한 비즈니스 로직**을 사용할 수 있습니다. UI 계층만 플랫폼별로 구현하여 네이티브 성능과 사용자 경험을 제공합니다.
