## 1. Unit of Work 패턴 (SQLDelight 기반)

### ✅ 권장 구현

```kotlin
// core-common/UnitOfWork.kt
interface UnitOfWork {
    suspend fun  transaction(block: suspend () -> T): Result
}

// core-diary-data/SqlDelightUnitOfWork.kt
class SqlDelightUnitOfWork(
    private val driver: SqlDriver
) : UnitOfWork {
    
    override suspend fun  transaction(block: suspend () -> T): Result =
        withContext(Dispatchers.IO) {
            runCatching {
                driver.transactionWithResult {
                    runBlocking { block() }
                }
            }
        }
}

// 사용 예시
class DiaryService(
    private val unitOfWork: UnitOfWork,
    private val diaryRepo: DiaryRepository,
    private val flowerRepo: FlowerRepository
) {
    suspend fun createDiaryWithFlower(diary: Diary, flower: Flower): Result =
        unitOfWork.transaction {
            diaryRepo.save(diary)
            flowerRepo.save(flower)
        }
}
```

## 2. SOLID 원칙 준수 패턴

```kotlin
// ✅ 단일 책임
class DiaryValidator {
    fun validate(diary: Diary): ValidationResult {
        return when {
            diary.title.isBlank() -> ValidationResult.Error("제목은 필수입니다")
            diary.content.length > MAX_CONTENT_LENGTH -> ValidationResult.Error("내용이 너무 깁니다")
            else -> ValidationResult.Success
        }
    }
    
    companion object {
        private const val MAX_CONTENT_LENGTH = 1000
    }

class DiaryFormatter {

            formattedDate = formatDate(diary.createdAt)
        )
    }
}
```

### 예외 처리 모듈화
```kotlin
// core-common/ExceptionUtil.kt
object ExceptionUtil {
    
    inline fun  safeCall(
        action: () -> T,
        onError: (Throwable) -> T
    ): T = try {
        action()
    } catch (e: Exception) {
        onError(e)
    }
    
    suspend inline fun  safeSuspendCall(
        crossinline action: suspend () -> T,
        crossinline onError: suspend (Throwable) -> T
    ): T = try {
        action()
    } catch (e: Exception) {
        onError(e)
    }
}

// 사용 예시
class DiaryRepository {
    suspend fun findById(id: DiaryId): Result =
        ExceptionUtil.safeSuspendCall(
            action = { queries.selectById(id.value).executeAsOneOrNull() },
            onError = { Result.failure(it) }
        ).let { Result.success(it) }
}
```
